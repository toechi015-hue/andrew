<script>
const menu = [
  {name:"Burger", price:12, img:"https://images.unsplash.com/photo-1568901346375-23c9450c58cd"},
  {name:"Pizza", price:15, img:"https://images.unsplash.com/photo-1601924928376-7b1e9e6d2c2e"},
  {name:"Pasta", price:10, img:"https://images.unsplash.com/photo-1589308078055-eb4c4c2fcd1a"},
  {name:"Salad", price:8, img:"https://images.unsplash.com/photo-1550304943-4f24f54ddde9"}
];

const menuContainer = document.getElementById("menuItems");
const cartItems = document.getElementById("cartItems");
const totalEl = document.getElementById("total");

let cart = [];

/* Load menu */
menu.forEach((item, index) => {
  menuContainer.innerHTML += `
    <div class="card">
      <img src="${item.img}">
      <div class="card-content">
        <h3>${item.name}</h3>
        <p class="price">$${item.price}</p>
        <button class="btn" onclick="addToCart(${index})">Add</button>
      </div>
    </div>
  `;
});

/* Add to cart */
function addToCart(index) {
  cart.push(menu[index]);
  renderCart();
}

/* Render cart */
function renderCart() {
  cartItems.innerHTML = "";
  let total = 0;

  cart.forEach(item => {
    total += item.price;
    cartItems.innerHTML += `<p>${item.name} - $${item.price}</p>`;
  });

  totalEl.innerText = "Total: $" + total;
}

/* Toggle cart */
function toggleCart() {
  const cartDiv = document.getElementById("cart");
  cartDiv.style.display = cartDiv.style.display === "none" ? "block" : "none";
}
</script>
